  def get_fields_for_requestdef( request_def )
    
    # puts "at start of get_fields for requestdef"
    
    fields_hash = {}
    
    request_def.fields.each do |f|
      fields_hash.merge!({f.field_name => f.field_label})
    end    
    
    return fields_hash
    
  end
