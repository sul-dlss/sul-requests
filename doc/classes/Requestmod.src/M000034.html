  def add_items( parm_list, items )
    
    require 'cgi'
    
    item_string = ''
    
    items.each do |item| 
      item.gsub!(/\n+/, " ")
      item.gsub!(/\s+/, " ")
      item.strip!
      if item_string == ''
        item_string = item_string + CGI::escape(item)  
      else
        item_string = item_string + '%5E' +CGI::escape(item)        
      end 
    end
      
      
    return parm_list + '&items=' + item_string  
    
  end
